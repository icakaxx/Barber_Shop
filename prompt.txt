            nextStep();
        }

        function setDateTime(time) {
            bookingState.time = time;
            bookingState.date = new Date().toLocaleDateString('en-GB', { weekday: 'short', day: 'numeric', month: 'short' });
            nextStep();
        }

        function nextStep() {
            if (bookingState.step < 4) {
                bookingState.step++;
                renderStep();
            }
        }

        function prevStep() {
            if (bookingState.step > 1) {
                bookingState.step--;
                renderStep();
            }
        }

        function renderStep() {
            const content = document.getElementById('wizardContent');
            const indicator = document.getElementById('stepIndicator');
            const progress = document.getElementById('progressBar');
            const prevBtn = document.getElementById('prevBtn');
            const footerAction = document.getElementById('footerAction');

            indicator.textContent = `Step ${bookingState.step} of 4`;
            progress.style.width = `${(bookingState.step / 4) * 100}%`;
            prevBtn.classList.toggle('invisible', bookingState.step === 1);
            footerAction.innerHTML = '';

            let html = '';

            if (bookingState.step === 1) {
                html = `<h3 class="text-xl font-bold mb-6">Choose a Service</h3><div class="space-y-3">${services.map(s => `<button onclick="setService('${s.id}')" class="w-full p-4 border rounded-xl text-left flex justify-between items-center transition-all hover:border-black group ${bookingState.service?.id === s.id ? 'border-black bg-black text-white' : 'border-gray-200'}"><div><p class="font-bold">${s.name}</p><p class="text-sm ${bookingState.service?.id === s.id ? 'text-gray-400' : 'text-gray-500'}">${s.duration}</p></div><div class="text-right"><p class="font-bold">${s.price}</p>${s.best ? `<span class="text-[10px] bg-white text-black px-1.5 py-0.5 rounded font-bold uppercase">Best Value</span>` : ''}</div></button>`).join('')}</div>`;
            } else if (bookingState.step === 2) {
                html = `<h3 class="text-xl font-bold mb-6">Select a Barber</h3><div class="grid grid-cols-1 sm:grid-cols-2 gap-4">${barbers.map(b => `<button onclick="setBarber('${b.id}')" class="p-4 border rounded-xl text-center transition-all hover:border-black ${bookingState.barber?.id === b.id ? 'border-black bg-black text-white' : 'border-gray-200'}"><div class="w-12 h-12 bg-gray-100 rounded-full mx-auto mb-3 flex items-center justify-center text-black"><i data-lucide="user" class="w-6 h-6"></i></div><p class="font-bold">${b.name}</p><p class="text-xs ${bookingState.barber?.id === b.id ? 'text-gray-400' : 'text-gray-500'}">${b.role}</p></button>`).join('')}</div>`;
            } else if (bookingState.step === 3) {
                html = `<h3 class="text-xl font-bold mb-2">Pick a Time</h3><p class="text-sm text-gray-500 mb-6">Availability for Today</p><div class="grid grid-cols-3 gap-3">${['09:00', '09:30', '10:00', '11:00', '13:30', '14:00', '15:30', '16:00', '17:30'].map(t => `<button onclick="setDateTime('${t}')" class="py-3 border border-gray-200 rounded-lg text-center font-medium hover:border-black hover:bg-black hover:text-white transition-all">${t}</button>`).join('')}</div>`;
            } else if (bookingState.step === 4) {
                html = `<h3 class="text-xl font-bold mb-6">Your Details</h3><div class="space-y-4"><div><label class="block text-sm font-medium text-gray-700 mb-1">Full Name</label><input type="text" id="custName" class="w-full p-3 border border-gray-200 rounded-lg focus:ring-2 focus:ring-black focus:outline-none" placeholder="John Doe"></div><div><label class="block text-sm font-medium text-gray-700 mb-1">Phone Number</label><input type="tel" id="custPhone" class="w-full p-3 border border-gray-200 rounded-lg focus:ring-2 focus:ring-black focus:outline-none" placeholder="+359 ..."></div><div class="bg-gray-50 p-4 rounded-xl border border-gray-100 mt-6"><p class="text-xs font-bold text-gray-400 uppercase mb-2">Summary</p><div class="flex justify-between items-center mb-1"><span class="text-sm font-medium">${bookingState.service?.name}</span><span class="text-sm font-bold">${bookingState.service?.price}</span></div><div class="text-xs text-gray-500">${bookingState.barber?.name} • ${bookingState.date} at ${bookingState.time}</div></div></div>`;
                footerAction.innerHTML = `<button onclick="confirmBooking()" class="bg-black text-white px-8 py-2 rounded-lg font-bold hover:bg-black/90 transition-colors">Confirm Appointment</button>`;
            }

            content.innerHTML = html;
            lucide.createIcons();
        }

        function confirmBooking() {
            const name = document.getElementById('custName').value;
            const phone = document.getElementById('custPhone').value;
            if (!name || !phone) { alert('Please fill in your details.'); return; }
            const content = document.getElementById('wizardContent');
            const footer = document.querySelector('#bookingModal .border-t');
            const stepper = document.querySelector('#bookingModal .h-1\\.5');
            const indicator = document.getElementById('stepIndicator');
            stepper.classList.add('hidden');
            footer.classList.add('hidden');
            indicator.textContent = 'Success!';
            content.innerHTML = `<div class="text-center py-12 animate-in slide-in-from-bottom-4 duration-500"><div class="w-20 h-20 bg-green-100 text-green-600 rounded-full flex items-center justify-center mx-auto mb-6"><i data-lucide="check" class="w-10 h-10"></i></div><h3 class="text-3xl font-bold mb-2">Confirmed!</h3><p class="text-gray-500 mb-8 px-8">Your appointment is booked. We've sent the details to your phone.</p><button onclick="closeBooking()" class="w-full py-4 bg-black text-white rounded-xl font-bold hover:bg-black/90 transition-all">Got it</button></div>`;
            lucide.createIcons();
        }

        // --- Year Helper ---
        document.getElementById('year').textContent = new Date().getFullYear();

        // Attach event listeners to all "Schedule" buttons
        document.querySelectorAll('a[href="#"]').forEach(btn => {
            if (btn.textContent.trim().toLowerCase().includes('schedule') || 
                btn.textContent.trim().toLowerCase().includes('book') || 
                btn.textContent.trim().toLowerCase().includes('now')) {
                btn.onclick = (e) => {
                    e.preventDefault();
                    if(currentView === 'landing') openBooking();
                };
            }
        });
    </script>
